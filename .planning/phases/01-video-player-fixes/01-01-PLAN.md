**Step 2 — Rewrite `openVideo()`** to branch based on `isMobile()`:

```javascript
window.openVideo = function(videoId) {
    const modal = document.getElementById('videoModal');
    modal.classList.add('active');
    document.body.style.overflow = 'hidden';

    if (isMobile()) {
        // Mobile: show YouTube iframe, hide Vidstack player
        const content = modal.querySelector('.video-modal-content');
        content.querySelector('media-player').style.display = 'none';
        let iframe = content.querySelector('.yt-mobile-iframe');
        if (!iframe) {
            iframe = document.createElement('iframe');
            iframe.className = 'yt-mobile-iframe';
            iframe.setAttribute('allowfullscreen', '');
            iframe.setAttribute('allow', 'autoplay; encrypted-media');
            iframe.setAttribute('frameborder', '0');
            content.appendChild(iframe);
        }
        iframe.src = `https://www.youtube.com/embed/${videoId}?autoplay=1`;
        iframe.style.display = '';
    } else {
        // Desktop: use Vidstack
        const content = modal.querySelector('.video-modal-content');
        const mobileIframe = content.querySelector('.yt-mobile-iframe');
        if (mobileIframe) {
            mobileIframe.style.display = 'none';
            mobileIframe.src = '';
        }
        content.querySelector('media-player').style.display = '';

        if (player) {
            firstPlay = true;
            const onCanPlay = () => {
                player.play();
                player.removeEventListener('can-play', onCanPlay);
            };
            player.addEventListener('can-play', onCanPlay);
            player.src = `youtube/${videoId}`;
        }
    }
}
```

**Step 3 — Rewrite `closeModal()`** to also clean up the mobile iframe:

```javascript
window.closeModal = function(event) {
    if (event && event.target !== event.currentTarget) return;
    const modal = document.getElementById('videoModal');
    modal.classList.remove('active');
    document.body.style.overflow = '';

    // Stop mobile iframe
    const mobileIframe = modal.querySelector('.yt-mobile-iframe');
    if (mobileIframe) {
        mobileIframe.src = '';
    }

    // Stop Vidstack player
    if (player) {
        player.pause();
        setTimeout(() => {
            player.src = '';
        }, 500);
    }
}
```

/* DSK-01/DSK-02: Hide duplicate loading spinner — keep Vidstack's built-in player-level spinner,

.video-modal-content media-video-layout .vds-buffering-indicator {

This targets only the spinner rendered inside `<media-video-layout>`, not the one rendered by `<media-player>` directly. Vidstack's base player spinner (outside the layout component) remains visible and functional.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User → YouTube embed | The videoId parameter from data.json is interpolated into an iframe src URL |
| data.json → openVideo() | videoId values from the JSON data file are used directly in URLs and Vidstack src strings |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Tampering | `openVideo(videoId)` — videoId injected into iframe src | accept | videoId values come from the site's own `data.json`, not user input. No form field or URL parameter feeds this value. The attack surface is the static JSON file itself, which requires repo write access to tamper with. |
| T-01-02 | Information Disclosure | YouTube iframe autoplay | accept | Autoplay is an intentional product decision (MOB-02). YouTube's embed is sandboxed by the browser's iframe sandbox model. No site data is exposed to the iframe. |
| T-01-03 | Denial of Service | Vidstack CDN unavailability | accept | Low-value target; CDN (jsDelivr) has high availability SLA. Desktop fallback to broken player on CDN failure is acceptable for a festival info site. |
</threat_model>
